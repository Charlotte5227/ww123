// ========================================== // 初期データの自動読み込み (map-data.json) // ========================================== async function loadInitialData() { try { // 同じディレクトリにある map-data.json を取得 // ※ URLが異なる場合は適宜書き換えてください（例: "./data/map-data.json"） const response = await fetch('map-data.json'); // ファイルが存在しない場合などは処理を中断 if (!response.ok) { console.log("map-data.json が見つからないか、取得できませんでした。"); return; } const data = await response.json(); // データが配列形式であれば読み込みを実行 if (Array.isArray(data)) { importData(data); console.log("初期データの読み込みが完了しました。"); } } catch (e) { console.error("初期データの読み込み中にエラーが発生しました:", e); } } // HTMLの読み込みが完了したら自動で実行する window.addEventListener("DOMContentLoaded", loadInitialData);
